def likelihood(D,p):
    N= D.size; suc= D.sum()
    theta=np.linspace(0,1,100)
    return p**suc*(1-p)**(N-suc)

def likelihood(D,T,scale):
    N=D.size; sr=(1/scale)
    y_bar=D.sum()/N
    mle=[]
    for i in T:
        y_in=[]
        for j in D:
            y_in.append((-(j-i)**2)*(sr**2)/2)
            y_sum = sum(y_in)
        mle.append(sr*np.exp(y_sum))
    return mle

Likelihood¶

In [6]:
def likelihood(D, p):
    N = D.size
    suc = D.sum()
    theta = np.linspace(0, 1, 100)
    return (p ** suc) * ((1 - p) ** (N - suc))

def log_likelihood(d, *, x):
    res = []
    for mu in x:
        res.append(-np.square((d - mu).sum()/2))    
    return res

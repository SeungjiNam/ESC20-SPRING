Compare MSE of test and training dataset¶

In [19]:
train_mse=[]
test_mse=[]
for i in np.arange(2,13):
    poly = PolynomialFeatures(degree=i)
    x_train_poly = poly.fit_transform(x_train)
    x_test_poly = poly.fit_transform(x_test)
    
    lin=LinearRegression(fit_intercept=False)
    lin.fit(x_train_poly, y_train)
    
    train_prediction=lin.predict(x_train_poly)
    tr_mse = math.log(sum(np.array(train_prediction-y_train)**2)/train_df.shape[0])
    train_mse.append(tr_mse)
    
    test_prediction=lin.predict(x_test_poly)
    ts_mse = math.log(sum(np.array(test_prediction-y_test)**2)/train_df.shape[0])
    test_mse.append(ts_mse)
print(train_mse)    
print(test_mse)

plt.plot(np.arange(2,13), train_mse, marker='o')
plt.plot(np.arange(2,13), test_mse, marker='o')
plt.legend(['Train MSE', 'Test MSE'])

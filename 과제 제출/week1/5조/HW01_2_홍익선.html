Bias-Variance Trade off

In [338]:
train_mse=[]
test_mse=[]

for i in np.arange(1, 13):
    poly = PolynomialFeatures(degree=i)
    X_train_poly = poly.fit_transform(X_train)
    X_test_poly = poly.fit_transform(X_test)
    lin3 = LinearRegression(fit_intercept=False)
    lin3.fit(X_train_poly, y_train)
    train_pred = lin3.predict(X_train_poly)
    train_mse.append(np.array(np.square(train_pred - y_train)).sum()/X_train_poly.shape[0])
    
    test_pred = lin3.predict(X_test_poly)
    test_mse.append(np.array(np.square(test_pred - y_test)).sum()/X_test_poly.shape[0])
    
plt.plot(np.arange(1,13), train_mse, marker='o')
plt.plot(np.arange(1,13), test_mse, marker='o')
plt.legend(['Train MSE', 'Test MSE'])

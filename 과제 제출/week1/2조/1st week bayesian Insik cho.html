## Likelihood

def likelihood(D,p):
    N = D.size; suc = D.sum()
    theta = np.linspace(0,1,100)
    return p**suc * (1-p)**(N - suc)

plt.plot(theta, likelihood(data, theta), color = 'g')
plt.title("likelihood")
plt.xlabel("theta")
plt.ylabel("p(Data|theta)")
plt.ylim(0,0.0005)

def likelihood(x, mu):
    like = 0 
    for i in x:
        like += np.log(norm.pdf(i, mu, 1))
    return like

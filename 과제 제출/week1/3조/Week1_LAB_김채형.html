# scipy stats packakge does not allow for plotting pdf-theta so we need to define formula directly.
def likelihood(D, p):
    N=D.size
    suc=D.sum()
    theta=np.linspace(0,1,100)
    return p**suc*(1-p)**(N-suc)

In [18]:
def likelihood(Data, mu, sd):
    loglikelihood=0
    for i in range(len(Data)):
        loglikelihood=loglikelihood+np.log((1/(np.sqrt(2*np.pi)*sd))*np.exp(-((Data[i]-mu)**2)/(2*sd**2)))
    return loglikelihood

plt.plot(theta, likelihood(data, theta, sd), color='g')
plt.axvline(x=theta[np.where(likelihood(data, theta, sd)==max(likelihood(data, theta, sd)))], linestyle=':')
plt.title('log likelihood')
plt.xlabel('mu')

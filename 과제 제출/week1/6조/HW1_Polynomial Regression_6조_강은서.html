for i in range(0,12):
    
    poly = PolynomialFeatures(degree=i+1)
    X_train_poly = poly.fit_transform(X_train)
    X_test_poly = poly.fit_transform(X_test)
    
    model = LinearRegression(fit_intercept=False)
    model.fit(X_train_poly, y_train) #model에 fitting된 모델 저장
    
    y_train_hat = model.predict(X_train_poly)
    residual = np.array(y_train) - y_train_hat 
    mse_train[i] = np.mean(residual**2)
    
    y_test_hat = model.predict(X_test_poly)
    residual_test = np.array(y_test) - y_test_hat
    mse_test[i] = np.mean(residual_test**2)

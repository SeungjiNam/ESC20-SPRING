mse_train=np.zeros(12)
mse_test=np.zeros(12)
for i in range(1,13):
    poly=PolynomialFeatures(degree=i) # 1~12
    X_train_poly=poly.fit_transform(X_train)
    X_test_poly=poly.fit_transform(X_test)
    
    lin=LinearRegression(fit_intercept=False) #design matrix에 이미 b0 포함
    lin.fit(X_train_poly,y_train)
    
    y_train_hat=lin.predict(X_train_poly)
    residual_train=np.array(y_train)-y_train_hat
    mse_train[i-1]=np.log(np.mean(residual_train**2))
    
    y_test_hat=lin.predict(X_test_poly)
    residual_test=np.array(y_test)-y_test_hat
    mse_test[i-1]=np.log(np.mean(residual_test**2))
    
print(mse_train,mse_test)

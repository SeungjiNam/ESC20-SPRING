def loglikelihood(data, mu):
    datavalue = []
    loglikelihood = 0
    for i in range(len(data)):
            loglikelihood = loglikelihood + np.log(np.exp(-((data[i]-mu)**2)/18)*(1/(3*np.sqrt(2*np.pi))))
    return loglikelihood

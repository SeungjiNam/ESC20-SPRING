# scipy stats package does not allow for plotting pdf-theta so we need to define formula directly.
def likelihood(D, p):
    N = D.size; suc = D.sum()
    theta = np.linspace(0,1,100)
    return p**suc * (1-p)**(N-suc) # note that we neglected the constant as it will be canceled out in appling Bayes Rule

mse_train = np.zeros(12) ; mse_test = np.zeros(12)


for i in range(12):
    
    poly = PolynomialFeatures(degree=i+1)
    X_train_poly = poly.fit_transform(X_train)
    X_test_poly = poly.fit_transform(X_test)
    model = LinearRegression(fit_intercept=False)
    model.fit(X_train_poly, y_train)
    y_train_hat = model.predict(X_train_poly)
    residual = y_train - y_train_hat 
    mse_train[i] = np.mean(residual**2)

    y_test_hat = model.predict(X_test_poly)
    residual_test = y_test - y_test_hat
    mse_test[i] = np.mean(residual_test**2)

logmse_train = np.log(mse_train)
logmse_test  = np.log(mse_test)
